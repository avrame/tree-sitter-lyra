trait Functor {
  map: (Self<a>, (a) -> b) -> Self<b>
}

impl Functor for Array {
  map = (arr, fn) => {
    var result = []
    for item in arr {
      result = [...result, fn(item)]
    }
    result
  }
}

impl Functor for Maybe {
  map = (maybe, fn) => match maybe {
    Some(x) => Some(fn(x)),
    None => None
  }
}

trait Eq {
  (_==_): (Self, Self) -> Bool,
  (_!=_): (Self, Self) -> Bool
}

impl Eq for []t {
  (_==_) = (a, b) => {
    if a.length() != b.length() {
      false
    } else {
      for i in 0..<a.length() {
        if a[i] != b[i] {
          return false
        }
      }
      true
    }
  },
  (_!=_) = (a, b) => !(a == b)
}

  add: (c<e>, e) -> c<e>,
  remove: (c<e>, e) -> c<e>,
  contains: (c<e>, e) -> Bool,
  size: (c<e>) -> Int
}

impl Collection<Array, Str> for StringList {
  add = (arr, item) => [...arr, item],
  remove = (arr, item) => filter(arr, (x) => x != item),
  contains = (arr, item) => any(arr, (x) => x == item),
  size = (arr) => arr.length()
